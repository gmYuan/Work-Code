1 Q: 如何实现 数组按字母A-Z进行排序

1 A:

```js
_normalizeSinger(list) {

  let map = {   //S1 数组内 对象格式为 { A: {title: xxx, items: [xxx]} }
    hot: {
      title: HOT_NAME,
      items: []
    }
  }

  list.forEach((item, index) => {
    if (index < 10) {     //S2 取前10项为 热门Hot
      map.hot.items.push(new Singer({
        name: item.Fsinger_name,
        id: item.Fsinger_mid
      }))
    }
    const key = item.Findex  //S3 key = A/B/C...Z
    if (!map[key]) {
      map[key] = {
        title: key,
        items: []
      }
    }
    map[key].items.push(new Singer({
      name: item.Fsinger_name,
      id: item.Fsinger_mid
    }))
  })
  
  //S4 为了得到有序一维数组，我们需要处理 map
  let ret = []
  let hot = []
  for (let key in map) {
    let val = map[key]
    if (val.title.match(/[a-zA-Z]/)) {
      ret.push(val)
    } else if (val.title === HOT_NAME) {
      hot.push(val)
    }
  }
  
  // S5 把对象按A/B/C进行排序，知识点是: sort + charCodeAt
  ret.sort((a, b) => {
    return a.title.charCodeAt(0) - b.title.charCodeAt(0)
  })
  
  return hot.concat(ret)
}
```
